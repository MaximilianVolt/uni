% Function to redefine letters and numbers
\newcommand{\colorlettersandnumbers}{
  % Letters (A-Z and a-z)
  \count@=`A
  \loop
    \begingroup
      \lccode`~=\count@
      \lowercase{\endgroup
      \def~{{\textcolor{cVariable}{\char\count@}}}}
    \mathcode\count@="8000
    \advance\count@ by 1
  \ifnum\count@<`Z+1
  \repeat
  
  \count@=`a
  \loop
    \begingroup
      \lccode`~=\count@
      \lowercase{\endgroup
      \def~{{\textcolor{cVariable}{\char\count@}}}}
    \mathcode\count@="8000
    \advance\count@ by 1
  \ifnum\count@<`z+1
  \repeat

  % Numbers (0-9)
  \count@=`0
  \loop
    \begingroup
      \lccode`~=\count@
      \lowercase{\endgroup
      \def~{{\textcolor{cLiteral}{\char\count@}}}}
    \mathcode\count@="8000
    \advance\count@ by 1
  \ifnum\count@<`9+1
  \repeat
}
\makeatother
